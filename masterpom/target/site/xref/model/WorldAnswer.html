11  public class WorldAnswer {
12  	private Iworld world ;
13  	
14  	/**
15  	 * Constructor of WorldAnswer
16  	 * @param world
17  	 * 			the world
18  	 */
19  	public WorldAnswer(World world){
20  		this.world = world ;
21  	}
22  	
23  	/**
24  	 * Get the world
25  	 * @return world
26  	 */
27  	public Iworld getWorld(){
28  		return this.world ;
29  	}
30  	
31  	/**
32  	 * Get the answer of the world
33  	 */
34  	public void getWorldAnswer(){
35  		final IElementsActionOnHeroes element = this.getWorld().getElementXY(this.getWorld().getLorann().getX(), this.getWorld().getLorann().getY());
36  
37  		switch (element.getElementActionOnHeroes()) {
38  			case COLLECT:
39  				System.out.println("COLLECT !!!! (:");
40  				this.resolveCollectTreasure((Itreasures) element);
41  				break;
42  			case NEWLIFE:
43  				System.out.println("NEW LIFE !!!! (:");
44  				this.resolveObtainNewLife();
45  				break;
46  			case UNLOCKS_GATE:
47  				System.out.println("GATE --> OPEN !!!! (:");
48  				this.resolveUnlocksGate();
49  				break;
50  			case EXIT:
51  				System.out.println("LEVEL --> END !!!! (:");
52  				this.resolveEndLevel();
53  				break;
54  			default:
55  				break;
56  		}
57  		this.getWorld().worldHasChanged() ;
58  	}
59  	
60  	/**
61  	 * Method execute when the sprite is a treasure
62  	 * @param element
63  	 * 			the element
64  	 */
65  	private void resolveCollectTreasure(Itreasures element){
66  		System.out.print("score : " + this.getWorld().getLorann().getScore() + " | new ");
67  		final int score = this.getWorld().getLorann().getScore() + element.collectTreasure() ;
68  		this.getWorld().getLorann().setScore(score) ;
69  		System.out.println("Score : " + this.getWorld().getLorann().getScore());
70  		this.getWorld().dropElement(this.getWorld().getLorann().getX(), this.getWorld().getLorann().getY());
71  	}
72  	
73  	/**
74  	 * Method execute when Lorann pick up the idol
75  	 */
76  	private void resolveObtainNewLife(){
77  		System.out.print("Lifes : " + this.getWorld().getLorann().getLife() + " | new ");
78  		final int life = this.getWorld().getLorann().getLife() + 1 ;
79  		this.getWorld().getLorann().setLife(life) ;
80  		System.out.println("Lifes : " + this.getWorld().getLorann().getLife());
81  		this.getWorld().dropElement(this.getWorld().getLorann().getX(), this.getWorld().getLorann().getY());
82  	}
83  	
84  	/**
85  	 * Method which unlock the gate and drop the blocs when Lorann pick up the Crystal-ball
86  	 */
87  	private void resolveUnlocksGate(){
88  		this.getWorld().dropElement(this.getWorld().getLorann().getX(), this.getWorld().getLorann().getY());
89  		this.getWorld().searchGate() ;
90  	}
91  	
92  	/**
93  	 * Method execute when Lorann is on the Gate-open, switch level
94  	 */
95  	private void resolveEndLevel(){
96  		this.getWorld().endLevel();
97  	}
98  }
